```glsl
precision mediump float;
uniform vec2 u_res;
uniform float u_time;
uniform float u_wave;
uniform float u_strength;
uniform vec2 u_origin;

// fBm noise using sin/cos approximation
float hash(vec2 p) {
  return fract(sin(dot(p, vec2(127.1, 311.7))) * 43758.5453);
}

float noise(vec2 p) {
  vec2 i = floor(p);
  vec2 f = fract(p);
  f = f * f * (3.0 - 2.0 * f);
  float a = hash(i);
  float b = hash(i + vec2(1.0, 0.0));
  float c = hash(i + vec2(0.0, 1.0));
  float d = hash(i + vec2(1.0, 1.0));
  return mix(mix(a, b, f.x), mix(c, d, f.x), f.y);
}

float fBm(vec2 p) {
  float val = 0.0;
  float amp = 0.5;
  for (int i = 0; i < 5; i++) {
    val += amp * noise(p);
    p *= 2.0;
    amp *= 0.5;
  }
  return val;
}

// Background A: #020202 fill + grid lines (#0d0d0d, 1px, every 16px)
vec3 bgA(vec2 fragCoord) {
  vec3 col = vec3(0.008); // #020202
  vec2 grid = mod(fragCoord, 16.0);
  if (grid.x < 1.0 || grid.y < 1.0) {
    col = vec3(0.051); // #0d0d0d
  }
  return col;
}

// Background B: bgA grid + radial gradient rgba(5,15,60,0.4) -> transparent
vec3 bgB(vec2 fragCoord, vec2 center, float radius) {
  vec3 col = bgA(fragCoord);
  float d = length(fragCoord - center) / radius;
  float grad = 1.0 - smoothstep(0.0, 1.0, d);
  col = mix(col, vec3(0.02, 0.059, 0.235), grad * 0.4);
  return col;
}

void main() {
  vec2 fragCoord = gl_FragCoord.xy;
  vec2 uv = fragCoord / u_res;
  uv.x *= u_res.x / u_res.y; // aspect correction

  vec2 origin = u_origin;
  origin.x *= u_res.x / u_res.y; // same aspect correction for origin

  float dist = length(uv - origin);

  // If no wave active, just render bgA
  if (u_wave <= 0.0) {
    gl_FragColor = vec4(bgA(fragCoord), 1.0);
    return;
  }

  float organic = fBm(uv * 3.0);
  float noisyDist = dist + (organic - 0.5) * 0.15 * u_strength;

  float bandWidth = 0.12;
  float inBand = smoothstep(u_wave - bandWidth, u_wave, noisyDist)
               * (1.0 - smoothstep(u_wave, u_wave + 0.02, noisyDist));

  vec2 dir = normalize(uv - origin);
  vec2 perp = vec2(-dir.y, dir.x);

  float localNoise = noise(uv * 10.0 + u_time);

  // UV displacement
  vec2 dispUV = uv + dir * inBand * u_strength * 0.03 + perp * localNoise * 0.01;

  // Chromatic aberration: sample R, G, B at different scales
  float r = fBm(dispUV * 1.00);
  float g = fBm(dispUV * 0.82);
  float b = fBm(dispUV * 0.65);
  vec3 chromatic = vec3(r, g, b);

  // Lateral flare: magenta on +perp side, cyan on -perp side
  float perpDot = dot(uv - origin, perp);
  vec3 magenta = vec3(1.0, 0.0, 0.8); // #ff00cc
  vec3 cyan = vec3(0.0, 1.0, 0.933);  // #00ffee
  float flareStrength = inBand * u_strength * 0.5;
  chromatic += magenta * max(perpDot, 0.0) * flareStrength;
  chromatic += cyan * max(-perpDot, 0.0) * flareStrength;

  // Rim boost
  float rim = smoothstep(u_wave - 0.04, u_wave - 0.01, noisyDist);
  chromatic += vec3(0.8) * rim * u_strength * inBand;

  // Background blend: bgA outside bubble, bgB inside
  float viewportDiag = length(u_res);
  vec3 bgInside = bgB(fragCoord, u_res * u_origin, u_wave * viewportDiag);
  vec3 bgOutside = bgA(fragCoord);
  float inside = 1.0 - smoothstep(u_wave - 0.01, u_wave + 0.01, noisyDist);

  vec3 bg = mix(bgOutside, bgInside, inside);
  vec3 finalColor = bg + chromatic * inBand * u_strength;

  gl_FragColor = vec4(clamp(finalColor, 0.0, 1.0), 1.0);
}
```

**fBm for Canvas 2D (JS version):**
```javascript
function hashF(x, y) { return ((Math.sin(x * 127.1 + y * 311.7) * 43758.5453) % 1 + 1) % 1; }
function noiseF(x, y) {
  const ix = Math.floor(x), iy = Math.floor(y);
  const fx = x - ix, fy = y - iy;
  const sx = fx * fx * (3 - 2 * fx), sy = fy * fy * (3 - 2 * fy);
  const a = hashF(ix, iy), b = hashF(ix+1, iy), c = hashF(ix, iy+1), d = hashF(ix+1, iy+1);
  return a + (b-a)*sx + (c-a)*sy + (a-b-c+d)*sx*sy;
}
function fBmF(x, y, octaves) {
  let val = 0, amp = 0.5;
  for (let i = 0; i < octaves; i++) { val += amp * noiseF(x, y); x *= 2; y *= 2; amp *= 0.5; }
  return val;
}
```

**5. Audio (Web Audio API, lazy init):**
```javascript
let audioCtx = null;
function playBeep() {
  try {
    if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    if (audioCtx.state === 'suspended') audioCtx.resume();
    const osc = audioCtx.createOscillator();
    const gain = audioCtx.createGain();
    osc.type = 'sine';
    osc.frequency.setValueAtTime(880, audioCtx.currentTime);
    gain.gain.setValueAtTime(0.3, audioCtx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.08);
    osc.connect(gain);
    gain.connect(audioCtx.destination);
    osc.start(audioCtx.currentTime);
    osc.stop(audioCtx.currentTime + 0.08);
  } catch (e) { /* fail silently */ }
}
```

**7. Dormant loop start/stop:**

```javascript
let dormantRafId = null;
function startDormantLoop() {
  if (dormantRafId) return;
  // Initialize halos
  const now = performance.now() * 0.001;
  halos = [{ start: now }, { start: now - 1 }, { start: now - 2 }];
  function loop(ts) {
    drawDormant(ts);
    dormantRafId = requestAnimationFrame(loop);
  }
  dormantRafId = requestAnimationFrame(loop);
}
function stopDormantLoop() {
  if (dormantRafId) { cancelAnimationFrame(dormantRafId); dormantRafId = null; }
}
```
