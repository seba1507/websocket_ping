**JavaScript (all inline):**

```javascript
// --- roomId from URL ---
const roomId = new URLSearchParams(window.location.search).get('room');
const connectScreen = document.getElementById('connecting-screen');
const errorScreen = document.getElementById('error-screen');
const gameScreen = document.getElementById('game-screen');
const pingBtn = document.getElementById('ping-btn');
const btnLabel = document.getElementById('btn-label');
const roomLabel = document.getElementById('room-label');
const cornerBr = document.getElementById('corner-br');

let pingTimestamp = null;

if (!roomId) {
  connectScreen.style.display = 'none';
  errorScreen.style.display = 'block';
  errorScreen.textContent = 'Error: No room ID in URL.';
  // Terminal state — no socket init
} else {
  const socket = io({ transports: ['websocket'] });

  // On connect (and reconnect), join room
  socket.on('connect', () => {
    socket.emit('phone:join', { roomId });
  });

  // server:joined -> State 2
  socket.on('server:joined', () => {
    connectScreen.style.display = 'none';
    errorScreen.style.display = 'none';
    gameScreen.style.display = 'flex';
    cornerBr.style.display = 'flex';
    roomLabel.textContent = 'ROOM \u00B7 ' + roomId;
  });

  // server:error -> terminal
  socket.on('server:error', ({ message }) => {
    connectScreen.style.display = 'none';
    gameScreen.style.display = 'none';
    cornerBr.style.display = 'none';
    errorScreen.style.display = 'block';
    errorScreen.textContent = 'Error: ' + message;
  });

  // disconnect -> back to State 1
  socket.on('disconnect', () => {
    // Reset in-flight state if pending
    if (pingBtn.disabled) {
      pingTimestamp = null;
      pingBtn.disabled = false;
      pingBtn.style.opacity = '1';
      pingBtn.style.pointerEvents = 'auto';
    }
    gameScreen.style.display = 'none';
    cornerBr.style.display = 'none';
    connectScreen.style.display = 'block';
  });

  // --- Press sequence ---
  let ghostTimeout = null;

  function handlePress(event) {
    if (pingBtn.disabled) return;

    // Store timestamp and disable
    pingTimestamp = Date.now();
    pingBtn.disabled = true;
    pingBtn.style.pointerEvents = 'none';
    socket.emit('phone:ping', { roomId });

    // Step 1: Impact flash (0ms, 80ms)
    pingBtn.style.transform = 'scale(0.96)';
    pingBtn.style.background = 'rgba(255,255,255,0.10)';
    setTimeout(() => {
      pingBtn.style.transform = '';
      pingBtn.style.background = 'rgba(255,255,255,0.04)';
    }, 80);

    // Step 2: Internal ripple (0ms, 400ms)
    const rect = pingBtn.getBoundingClientRect();
    let rippleX, rippleY;
    if (event.clientX !== undefined && event.clientX !== 0) {
      rippleX = event.clientX - rect.left;
      rippleY = event.clientY - rect.top;
    } else if (event.touches && event.touches.length > 0) {
      rippleX = event.touches[0].clientX - rect.left;
      rippleY = event.touches[0].clientY - rect.top;
    } else {
      rippleX = rect.width / 2;
      rippleY = rect.height / 2;
    }
    const size = Math.max(rect.width, rect.height);
    const ripple = document.createElement('div');
    ripple.classList.add('ripple');
    ripple.style.width = size + 'px';
    ripple.style.height = size + 'px';
    ripple.style.left = (rippleX - size / 2) + 'px';
    ripple.style.top = (rippleY - size / 2) + 'px';
    pingBtn.appendChild(ripple);
    ripple.addEventListener('animationend', () => ripple.remove());

    // Step 3: Chromatic ghost (60ms offset, 300ms duration)
    setTimeout(() => {
      btnLabel.style.textShadow = '2px 0 rgba(255,0,60,0.85), -2px 0 rgba(0,240,255,0.85)';
      btnLabel.style.color = 'rgba(255,255,255,0.95)';
      ghostTimeout = setTimeout(() => {
        btnLabel.style.textShadow = '';
        btnLabel.style.color = 'rgba(255,255,255,0.65)';
        ghostTimeout = null;
      }, 300);
    }, 60);

    // In-flight state
    pingBtn.style.opacity = '0.3';
  }

  pingBtn.addEventListener('touchstart', (e) => { e.preventDefault(); handlePress(e); }, { passive: false });
  pingBtn.addEventListener('click', handlePress);

  // --- server:pong handler ---
  socket.on('server:pong', () => {
    if (!pingTimestamp) return;
    const latencyMs = Date.now() - pingTimestamp;
    socket.emit('phone:latency', { roomId, latencyMs });
    if (navigator.vibrate) navigator.vibrate(80);
    // Re-enable button
    pingBtn.disabled = false;
    pingBtn.style.opacity = '1';
    pingBtn.style.pointerEvents = 'auto';
  });
}
```
